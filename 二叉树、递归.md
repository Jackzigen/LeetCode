public void recur(int level, int param) { 
    // terminator 终结条件
    if (level > MAX_LEVEL) { 
    // process result 流程结果
       return; 
    } 
    // process current logic  当前逻辑过程
    process(level, param); 
    // drill down 向下递归

    // restore current status 恢复现状
}
