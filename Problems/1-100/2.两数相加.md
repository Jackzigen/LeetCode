[2.两数相加](https://leetcode-cn.com/problems/add-two-numbers/)

/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
/**
 * 思路：从头开始操作链表，遇到边界情况如超过10，就做特殊处理
 */
var addTwoNumbers = function(l1, l2) {
    var List = new ListNode(0)
    var head = List
    var sum = 0
    var carry = 0

    while(l1 !== null || l2 !== null || sum > 0) {
         if(l1 !== null ) {
             sum = sum + l1.val
             l1 = l1.next
         }

         if(l2 !== null) {
             sum = sum + l2.val
             l2 = l2.next
         }

         if(sum >= 10){
             carry = 1
             sum = sum - 10
         }
         head.next = new ListNode(sum)
         head = head.next
         sum = carry
         carry = 0
    }

    return List.next

};
